iniciar todo desde cero
php artisan migrate:fresh

iniciar todo desde cero con seeders
php artisan migrate:fresh --seed


para generar modulos
primero
php artisan make:module Categoria \
--plural=categorias \
--fields="nombre:string(150):index, descripcion:string(255):nullable, status:boolean, id_establecimiento:foreignId:constrained=establecimientos:onDelete=cascade, id_user:foreignId:nullable:constrained=users:onDelete=set null" \
--force

segundo
php artisan migrate

para productos
php artisan make:module Producto --plural=productos --fields="codigo:string(200):index, descripcion:string(255):index, tipo_id:foreignId:constrained=tipos_producto, tarifa_iva_id:foreignId:constrained=tarifa_iva, precio_base:decimal(15,6), estado:string, id_establecimiento:foreignId:constrained=establecimientos:onDelete=cascade, id_user:foreignId:nullable:constrained=users:onDelete=set null" --force





seeders
php artisan db:seed
